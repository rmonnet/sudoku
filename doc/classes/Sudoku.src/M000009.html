    def initialize(steps, parent =nil)

        if parent then
            @problem = []
            parent.problem.each { |row| @problem.push row.dup }
            @steps = parent.steps.dup
        else
            @problem = []
            9.times { @problem.push Array.new(9, 0) }
            @steps = []
        end

        steps.each do |s|
            row, col, val = *s
            @problem[row][col] = val
            # only fill-in the list of solutions if this is not the
            # original problem
            @steps.push(Step.new(row, col, val)) if parent
        end
        validate
    end
