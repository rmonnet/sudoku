    def show_stats
        res = "\n"
        res += "#{@grid}\n"
        res += "initial problem + #{@steps.join(" + ")}\n"
        res += "max_poss         : #{max_poss}\n"
        res += "min_depth        : #{min_depth}\n"
        res += "degree_of_freedom: #{dof}\n"
        each_poss(min_depth) { |poss| res += "#{poss}\n" }
        res += "\n"
        res
    end
