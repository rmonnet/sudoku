    def analyze
        res = "\n"
        res +="poss:\n"
        @poss.each do |p|
            res += "  #{p}\n"
        end
        @rows.each_index do |r|
            res += "row##{r}: [#{@rows[r].to_a.sort.join(", ")}]\n"
        end
        @cols.each_index do |c|
            res += "col##{c}: [#{@cols[c].to_a.sort.join(", ")}]\n"
        end
        @box.each_index do |r|
            subrow = @box[r]
            subrow.each_index do |c|
                res += "box##{r}-#{c}: [#{subrow[c].to_a.sort.join(", ")}]\n"
            end
        end
        res
    end
