def onChore(sender, sel, ptr) 

    v = @table.getItemText(@current_r, @current_c).to_i
    @table.setItemText(@current_r, @current_c, "#{v+1}")
    @current_r += 1
    if @current_r > 8 then
        @current_r = 0
        @current_c = (@current_c + 1) % 9
    end
    getApp().addChore(method(:onChore)) if @running

end
