    def onCmdSetAsProblem(sender, sel, ptr)
        return if @running
        new_grid = []
        (0..8).step do |r|
            new_row = []
            (0..8).step do |c|
                v = @table.getItemText(r, c).to_i
                if v >= 1 and v <= 9 then
                    new_row.push(v)
                else
                    new_row.push(0)
                end
            end
            new_grid.push(new_row)
        end
        @initial_problem = Sudoku.new(new_grid)
        @solver= Solver.new(@initial_problem)
        fillGrid
    end
