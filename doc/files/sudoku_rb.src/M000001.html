def show_problem(p)
    puts
    puts p
    puts "initial problem + #{p.steps.join(" + ")}"
    puts "max_poss         : #{p.max_poss}"
    puts "min_depth        : #{p.min_depth}"
    puts "degree_of_freedom: #{p.dof}"
    p.each_poss(p.min_depth) { |poss| puts poss }
    puts
end
